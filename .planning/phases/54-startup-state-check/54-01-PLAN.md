**1. Add `httpGetJson(url)` helper** — Returns parsed JSON from a GET request. Uses the `http` module already imported in the file. Pattern consistent with existing `checkUrl()`:

```javascript
function httpGetJson(url) {
  return new Promise((resolve, reject) => {
    http.get(url, (res) => {
      let data = '';
      res.on('data', (chunk) => { data += chunk; });
      res.on('end', () => {
        try {
          resolve(JSON.parse(data));
        } catch {
          reject(new Error('Invalid JSON response'));
        }
      });
    }).on('error', reject);
  });
}
```

**2. Add `httpPost(url)` helper** — Returns HTTP status code from a POST request with empty body:

```javascript
function httpPost(url) {
  return new Promise((resolve, reject) => {
    const { hostname, port, pathname } = new URL(url);
    const req = http.request(
      { hostname, port: Number(port), path: pathname, method: 'POST',
        headers: { 'Content-Length': 0 } },
      (res) => resolve(res.statusCode)
    );
    req.on('error', reject);
    req.end();
  });
}
```

**3. Add `checkStartupState()` function** — Async function that determines which URL to load based on first-run state and license validity:

```javascript
// WIZD-01 / LICS-02 / LICS-04: Determine startup URL based on first-run state and license
async function checkStartupState() {
  const SETUP_URL = 'http://localhost:3000/setup';
  const APP_URL   = 'http://localhost:3000';

  try {
    // Step 1: Check first-run state (WIZD-01)
    const settingsData = await httpGetJson(
      'http://127.0.0.1:8000/api/v1/desktop/settings'
    );
    if (!settingsData || settingsData.first_run_complete !== true) {
      console.log('[launcher] First run detected — routing to setup wizard');
      return SETUP_URL;
    }

    // Step 2: Validate license on subsequent launches (LICS-02 / LICS-04)
    const licenseStatus = await httpPost(
      'http://127.0.0.1:8000/api/v1/desktop/license/validate'
    );
    if (licenseStatus === 200) {
      return APP_URL;
    }
    // 403 = expired/invalid, 404 = not activated (LICS-04)
    console.log(`[launcher] License check returned ${licenseStatus} — routing to setup`);
    return SETUP_URL;

  } catch (err) {
    // Network error — graceful degradation (backend has its own 7-day grace period)
    console.warn('[launcher] Startup state check failed (non-fatal):', err.message);
    return APP_URL;
  }
}
```

**CRITICAL anti-patterns to avoid:**
- Do NOT use `localhost` for API calls to backend — use `127.0.0.1` (avoids IPv6 mismatch on Windows per Phase 48 decision)
- Do NOT call license validate when first_run_complete is false — fresh install has no license yet
- Do NOT treat network errors as license failure — return APP_URL for graceful degradation
- Use `!== true` (not `=== false`) to catch undefined/null/missing key variants for first_run_complete
