**frontend/tests/e2e-library.spec.ts** — Library workflow E2E test:

```typescript
import { test, expect } from '@playwright/test';

test.describe('Library Workflow E2E', () => {

  test('library page loads and fetches projects from API', async ({ page }) => {
    // Intercept the API call the library page makes on mount
    const projectsResponse = page.waitForResponse(
      response => response.url().includes('/api/v1/library/projects')
        && response.request().method() === 'GET'
    );

    await page.goto('/library');
    await page.waitForLoadState('networkidle');

    // Wait for the API call and assert on the response
    const response = await projectsResponse;
    expect(response.status()).toBe(200);

    const data = await response.json();
    // Response should be an array (even if empty)
    expect(Array.isArray(data)).toBe(true);

    // Each project in the list should have expected fields
    if (data.length > 0) {
      const project = data[0];
      expect(project).toHaveProperty('id');
      expect(project).toHaveProperty('name');
      expect(project).toHaveProperty('status');
    }
  });

  test('library page fetches clips from API', async ({ page }) => {
    const clipsResponse = page.waitForResponse(
      response => response.url().includes('/api/v1/library/all-clips')
        && response.request().method() === 'GET'
    );

    await page.goto('/library');

    const response = await clipsResponse;
    expect(response.status()).toBe(200);

    const data = await response.json();
    expect(Array.isArray(data)).toBe(true);
  });

  test('health endpoint returns OK from frontend context', async ({ page }) => {
    // Direct API call through page.evaluate to verify backend connectivity
    const healthData = await page.evaluate(async () => {
      const res = await fetch('http://localhost:8000/api/v1/health');
      return { status: res.status, body: await res.json() };
    });

    expect(healthData.status).toBe(200);
    expect(healthData.body).toHaveProperty('status');
    expect(healthData.body.status).toBe('ok');
  });
});
```

import { test, expect } from '@playwright/test';

test.describe('Pipeline Workflow E2E', () => {

  test('pipeline page loads and fetches data from API', async ({ page }) => {
    // The pipeline page fetches scripts and/or segments on load
    // Intercept any API call to /api/v1/pipeline or /api/v1/segments
    const apiCalls: Array<{url: string, status: number, method: string}> = [];

    page.on('response', response => {
      if (response.url().includes('/api/v1/')) {
        apiCalls.push({
          url: response.url(),
          status: response.status(),
          method: response.request().method()
        });
      }
    });

    await page.goto('/pipeline');
    await page.waitForLoadState('networkidle');
    // Give a moment for async API calls
    await page.waitForTimeout(2000);

    // Verify at least one API call was made
    expect(apiCalls.length).toBeGreaterThan(0);

    // All API calls should return successful status codes (not 500)
    for (const call of apiCalls) {
      expect(call.status).toBeLessThan(500);
    }
  });

  test('pipeline page renders without crashing', async ({ page }) => {
    await page.goto('/pipeline');
    await page.waitForLoadState('domcontentloaded');

    // Page should not show an error boundary or blank screen
    const body = await page.textContent('body');
    expect(body).toBeTruthy();
    // Should not contain Next.js error messages
    expect(body).not.toContain('Application error');
    expect(body).not.toContain('Unhandled Runtime Error');
  });
});
```

import { test, expect } from '@playwright/test';

test.describe('Product Video Workflow E2E', () => {

  test('product video page loads and fetches products from API', async ({ page }) => {
    const apiCalls: Array<{url: string, status: number, body: any}> = [];

    page.on('response', async response => {
      if (response.url().includes('/api/v1/') && response.status() < 400) {
        try {
          const body = await response.json();
          apiCalls.push({
            url: response.url(),
            status: response.status(),
            body
          });
        } catch {
          // Non-JSON response, skip
          apiCalls.push({
            url: response.url(),
            status: response.status(),
            body: null
          });
        }
      }
    });

    await page.goto('/product-video');
    await page.waitForLoadState('networkidle');
    await page.waitForTimeout(2000);

    // Verify API calls were made
    expect(apiCalls.length).toBeGreaterThan(0);

    // Check that product/feed related API calls returned valid data
    const productCalls = apiCalls.filter(c =>
      c.url.includes('/products') || c.url.includes('/feed')
    );

    // If product calls were made, verify response structure
    for (const call of productCalls) {
      expect(call.status).toBe(200);
      if (call.body !== null) {
        // Response should be an array or object with expected shape
        expect(call.body).toBeDefined();
      }
    }
  });

  test('product video page renders without errors', async ({ page }) => {
    await page.goto('/product-video');
    await page.waitForLoadState('domcontentloaded');

    const body = await page.textContent('body');
    expect(body).toBeTruthy();
    expect(body).not.toContain('Application error');
    expect(body).not.toContain('Unhandled Runtime Error');
  });
});
```
